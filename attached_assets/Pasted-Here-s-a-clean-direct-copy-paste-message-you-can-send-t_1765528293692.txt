3. Add a permission guard component

With:

import { PERMISSIONS } from "../config/permissions";

export default function RequiresPermission({ userRole, feature, children }) {
  const allowed = PERMISSIONS[userRole]?.[feature];

  if (!allowed) {
    return (
      <div style={{ color: "#D4AF37", textAlign: "center", padding: 20 }}>
        Full access is only for focus-group members.  
        Join at NutriCoreApp.com.
      </div>
    );
  }

  return children;
}
